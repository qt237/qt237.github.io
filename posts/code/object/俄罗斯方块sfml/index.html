需要一个结构来存储方块

int block[7][4] = {1, 3, 5, 7, // I字型
3, 5, 4, 6, // Z字型
2, 4, 5, 7, // S字型2
, 3, 5, 7, // L字型
3, 5, 7, 6, // J字型
3, 5, 4, 7, // T字型
2, 3, 4, 5, // O字型};

这个转化为二维坐标x,y，就是x=block%2,y=block/2;

struct Point { int x; int y; } a[4], b[4];

a[4]来储存坐标

使用的话

    int n = 3;
    for (int i = 0; i < 4; i++) {
        a[i].x = block[n][i] % 2;
        a[i].y = block[n][i] / 2;
    }

旋转

        if (isRotate) {
            Point p = a[1];  // center
            for (int i = 0; i < 4; i++) {
                int x = a[i].y - p.y;
                int y = a[i].x - p.x;
                a[i].x = p.x - x;
                a[i].y = p.y + y;
            }
        }

时间控制自由下落

通过sf::Clock

            // float timer
            // sf::Clock clock;
            float time = clock.getElapsedTime().asSeconds();
            clock.restart();
            timer += time;
